   23 -- | Resource hash

   26 fetchStaticFile :: URL -> SHA256 -> (FilePath -> FilePath -> IO ()) -> IO FilePath
   27 fetchStaticFile url sha256 unpack = do
   28   root <- getReanimateCacheDirectory
   29   let folder = root </> sha256
   30   hit <- doesDirectoryExist folder
   31   unless hit $
   32     downloadFile url $ \path -> do
   33       inp <- B.readFile path
   34       let inpSha = B8.unpack (encode (hash inp))
   35       if inpSha == sha256
   36         then do
   37           unpack folder path
   38         else
   39           error $
   40             "URL " ++ url ++ "\n"
   41               ++ "  Expected SHA256: "
   42               ++ sha256
   43               ++ "\n"
   44               ++ "  Actual SHA256:   "
   45               ++ inpSha
   46   return folder

   50 -- | Download and unpack zip archive. The returned path is the unpacked folder.
   51 zipArchive :: URL -> SHA256 -> FilePath
   52 zipArchive url sha256 = unsafePerformIO $
   53   fetchStaticFile url sha256 $ \folder zipfile ->
   54     callProcess "unzip" ["-qq", "-d", folder, zipfile]
   55 
   56 {-# NOINLINE tarball #-}
   57 
   58 -- | Download and unpack tarball. The returned path is the unpacked folder.
   59 tarball :: URL -> SHA256 -> FilePath
   60 tarball url sha256 = unsafePerformIO $
   61   fetchStaticFile url sha256 $ \folder tarfile ->
   62     callProcess "tar" ["--overwrite", "--one-top-level=" ++ folder, "-xf", tarfile]
   63 
   64 downloadFile :: URL -> (FilePath -> IO a) -> IO a
   65 downloadFile url action = do
   66   mbCurl <- findExecutable "curl"
   67   mbWget <- findExecutable "wget"
   68   case (mbCurl, mbWget) of
   69     (Just curl, _) -> downloadFileCurl curl url action
   70     (_, Just wget) -> downloadFileWget wget url action
   71     (Nothing, Nothing) -> error "curl/wget required to download files"
   72 
   73 downloadFileCurl :: FilePath -> URL -> (FilePath -> IO a) -> IO a
   74 downloadFileCurl curl url action = withTempFile "dl" $ \path -> do
   75   callProcess
   76     curl
   77     [ url,
   78       "--output",
   79       path,
   80       "--silent",
   81       "--show-error",
   82       "--max-filesize",
   83       "10M"
   84     ]
   85   action path
   86 
   87 downloadFileWget :: FilePath -> URL -> (FilePath -> IO a) -> IO a
   88 downloadFileWget wget url action = withTempFile "dl" $ \path -> do
   89   callProcess
   90     wget
   91     [ url,
   92       "--output-document=" ++ path,
   93       "--quiet"
   94     ]
   95   action path
